\hline
\textbf{Mã UC} & UC08-04 \\
\hline
\textbf{Tên UC} & Xem xét Yêu cầu Trả hàng (Review Return Request) \\
\hline
\textbf{Mô tả tóm tắt} & Quản lý kho xem xét và phê duyệt hoặc từ chối một yêu cầu trả hàng. \\
\hline
\textbf{Tác nhân tham gia} & Warehouse Manager (Quản lý Kho) \\
\hline
\textbf{Sự kiện kích hoạt} & Quản lý nhận thông báo hoặc truy cập danh sách "RETURN_SUBMITTED". \\
\hline
\textbf{Luồng sự kiện chính} & 
\begin{enumerate}
    \vspace{-18pt}
    \item Quản lý mở yêu cầu "RETURN_SUBMITTED".
    \item Quản lý xem xét chi tiết, lý do, bằng chứng đính kèm.
    \item Quản lý nhấn "Phê duyệt" (Approve) hoặc "Từ chối" (Reject).
    \item Nếu "Phê duyệt", hệ thống cập nhật trạng thái "RETURN_APPROVED".
    \item Nếu "Từ chối", hệ thống yêu cầu nhập lý do và cập nhật trạng thái "RETURN_REJECTED".
\end{enumerate} \\
\hline
\textbf{Luồng sự kiện khác} & Từ chối: hệ thống yêu cầu nhập lý do từ chối (bắt buộc) và gửi thông báo. \\
\hline
\textbf{Yêu cầu trước khi thực hiện} & Yêu cầu trả hàng ở trạng thái "RETURN_SUBMITTED". \\
\hline
\textbf{Yêu cầu sau khi thực hiện} & Yêu cầu được cập nhật trạng thái (Approved/Rejected). \\
\hline
\textbf{Yêu cầu phi chức năng} & (Không có) \\
\hline
