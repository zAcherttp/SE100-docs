\hline
\textbf{Tên UC} & Xem Báo cáo Tồn kho (View Inventory Reports) \\
\hline
\textbf{Mô tả tóm tắt} & Xem các báo cáo chuẩn liên quan đến tình hình tồn kho. \\
\hline
\textbf{Tác nhân tham gia} & Warehouse Manager, Accountant, Report Viewer \\
\hline
\textbf{Sự kiện kích hoạt} & Người dùng chọn một báo cáo trong nhóm "Báo cáo Tồn kho". \\
\hline
\textbf{Luồng sự kiện chính} & 
\begin{enumerate}
    \vspace{-18pt}
    \item Người dùng chọn loại báo cáo (VD: Tồn kho, Định giá, Hết hạn, Chậm luân chuyển...).
    \item Người dùng nhập các bộ lọc (VD: Ngày, Kho, Danh mục).
    \item Nhấn "Xem Báo cáo".
    \item Hệ thống truy vấn dữ liệu và hiển thị kết quả dưới dạng bảng/biểu đồ.
\end{enumerate} \\
\hline
\textbf{Luồng sự kiện khác} & (Không có) \\
\hline
\textbf{Yêu cầu trước khi thực hiện} & Người dùng đã đăng nhập và có quyền "VIEW_REPORTS". \\
\hline
\textbf{Yêu cầu sau khi thực hiện} & Dữ liệu báo cáo được hiển thị. \\
\hline
\textbf{Yêu cầu phi chức năng} & Thời gian tạo báo cáo lớn (dữ liệu 1 năm) < 30 giây. \\
