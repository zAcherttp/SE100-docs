\textbf{Tên UC} & Từ chối Yêu cầu Điều chỉnh (Reject Adjustment Request) \\
\hline
\textbf{Mô tả tóm tắt} & Cấp có thẩm quyền từ chối một yêu cầu điều chỉnh. \\
\hline
\textbf{Tác nhân tham gia} & Warehouse Manager (Quản lý Kho), Inventory Administrator (có quyền APPROVE) \\
\hline
\textbf{Luồng sự kiện chính} & 
\begin{enumerate}
    \vspace{-18pt}
    \item Người phê duyệt xem xét yêu cầu.
    \item Người phê duyệt nhấn "Từ chối" (Reject).
    \item Hệ thống yêu cầu nhập "Lý do Từ chối" (bắt buộc).
    \item Người phê duyệt nhập lý do và xác nhận.
    \item Hệ thống cập nhật trạng thái "ADJ_REJECTED".
    \item Gửi thông báo (kèm lý do) cho người tạo.
\end{enumerate} \\
\hline
\textbf{Luồng sự kiện khác} & Không có \\
\hline
\textbf{Yêu cầu đặc biệt} & Phải bắt buộc nhập lý do từ chối. \\
\hline
\textbf{Trạng thái hệ thống ngay khi bắt đầu thực hiện} & Yêu cầu ở trạng thái "ADJ_PENDING_APPROVAL". \\
\hline
\textbf{Trạng thái hệ thống sau khi bắt đầu thực hiện} & 
\begin{enumerate}
    \vspace{-18pt}
    \item Yêu cầu có trạng thái "ADJ_REJECTED".
    \item Tồn kho không thay đổi.
\end{enumerate} \\

\textbf{Điểm mở rộng} & Không có \\
