\caption{Bảng UC01-12}
\begin{tabular}{|p{4cm}|p{10cm}|}
\hline
\textbf{Mã UC} & UC01-12 \\
\hline
\textbf{Tên UC} & Manage Users (Quản lý Người dùng) \\
\hline
\textbf{Mô tả tóm tắt} & Cho phép Quản trị viên xem, kích hoạt/vô hiệu hóa, và gán chi nhánh cho người dùng trong tổ chức. \\
\hline
\textbf{Tác nhân tham gia} & Quản trị viên hệ thống \\
\hline
\textbf{Luồng sự kiện chính} & 
\begin{enumerate}
    \vspace{-18pt}
    \item Admin điều hướng đến trang "Quản lý Người dùng".
    \item Hệ thống «include» View All Users (hiển thị danh sách tất cả người dùng trong tổ chức).
    \item Admin chọn một người dùng để quản lý.
    \item Admin thực hiện một trong các hành động:
    \item[4a.] Deactivate User (Vô hiệu hóa/Kích hoạt tài khoản).
    \item[4b.] Assign to Branches (Gán người dùng vào các chi nhánh cụ thể).
\end{enumerate} \\
\hline
\textbf{Luồng sự kiện khác} & Không có \\
\hline
\textbf{Yêu cầu đặc biệt} & Không thể xóa cứng (hard delete) người dùng để đảm bảo tính toàn vẹn của audit log. \\
\hline
\textbf{Trạng thái hệ thống ngay khi bắt đầu thực hiện} & Admin đã đăng nhập và có quyền "VIEW_USERS". \\
\hline
\textbf{Trạng thái hệ thống sau khi bắt đầu thực hiện} & Thông tin hoặc trạng thái của người dùng được cập nhật. \\
\hline
\textbf{Điểm mở rộng} & «extend» Soft Delete User (Xóa mềm người dùng). \\
\hline
\end{tabular}
\end{table}
