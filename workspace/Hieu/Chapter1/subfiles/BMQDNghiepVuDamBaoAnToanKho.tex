\subsubsection{Quy định (QĐ)}

\textbf{QĐ8: Quy định về An toàn kho}

\begin{itemize}
    \item \textbf{[PCCC]} Nghiêm cấm hút thuốc, sử dụng nguồn lửa trần trong khu vực kho.
    \item \textbf{[PCCC]} Lối đi lại và lối thoát hiểm phải luôn thông thoáng, không bị chặn.
    \item \textbf{[PCCC]} Hệ thống PCCC phải được kiểm tra định kỳ (theo BM17).
    \item \textbf{[An toàn Lao động]} Bắt buộc sử dụng trang thiết bị BHLĐ (theo BM19) khi làm việc trong kho.
    \item \textbf{[An toàn Lao động]} Tuân thủ quy trình vận hành an toàn các thiết bị nâng hạ (xe nâng...) và vỗ.
    \item \textbf{[An toàn Lao động]} Hàng hóa phải được sắp xếp trên kệ chắc chắn, đúng tải trọng, không đề vỗ.
    \item \textbf{[An ninh]} Hạn chế người không có phận sự ra vào khu vực kho.
    \item \textbf{[An ninh]} Mọi sự cố (tai nạn, cháy nổ, xâm nhập) phải được báo cáo ngay lập tức (theo BM18).
\end{itemize}
