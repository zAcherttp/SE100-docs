\caption{Đặc tả use case}
\begin{tabular}{|p{4cm}|p{10cm}|}
\hline
\textbf{Mã UC} & UC02-03 \\
\hline
\textbf{Tên UC} & Configure Category Settings (Cấu hình Cài đặt Danh mục) \\
\hline
\textbf{Mô tả tóm tắt} & Cho phép quản trị viên thiết lập các quy tắc kinh doanh cụ thể cho một danh mục (ví dụ: cảnh báo hết hạn). \\
\hline
\textbf{Tác nhân tham gia} & Quản lý kho / Quản trị viên hàng tồn kho \\
\hline
\textbf{Luồng sự kiện chính} & 
\begin{enumerate}
    \vspace{-18pt}
    \item Tác nhân điều hướng đến "Dữ liệu chủ" -> "Danh mục".
    \item Tác nhân chọn một danh mục và nhấn "Cài đặt".
    \item Hệ thống hiển thị biểu mẫu cài đặt (lưu trong category_settings).
    \item Tác nhân «include» Set Expiry Warning Thresholds (đặt ngưỡng cảnh báo hết hạn, ví dụ: 30, 60 ngày).
    \item Tác nhân «include» Set Default Storage Requirements (đặt yêu cầu lưu trữ mặc định, ví dụ: "Refrigerated").
    \item Tác nhân nhấn "Lưu".
    \item Hệ thống cập nhật các cài đặt (JSONB) cho danh mục.
\end{enumerate} \\
\hline
\textbf{Luồng sự kiện khác} & Không có \\
\hline
\textbf{Yêu cầu đặc biệt} & Các cài đặt này sẽ được áp dụng cho tất cả sản phẩm thuộc danh mục này, trừ khi bị ghi đè ở cấp sản phẩm. \\
\hline
\textbf{Trạng thái hệ thống ngay khi bắt đầu thực hiện} & Tác nhân đã đăng nhập và có quyền "MANAGE_CATEGORIES". \\
\hline
\textbf{Trạng thái hệ thống sau khi bắt đầu thực hiện} & Các quy tắc kinh doanh cho danh mục được cập nhật. \\
\hline
\textbf{Điểm mở rộng} & Không có \\
\hline
\end{tabular}
\end{table}
