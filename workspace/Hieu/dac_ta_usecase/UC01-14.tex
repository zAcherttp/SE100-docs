\hline
\textbf{Mã UC} & UC01-14 \\
\hline
\textbf{Tên UC} & Assign Roles to Users (Gán vai trò cho Người dùng) \\
\hline
\textbf{Mô tả tóm tắt} & Cho phép Quản trị viên gán các vai trò đã định nghĩa cho người dùng, áp dụng trên toàn tổ chức hoặc chi nhánh cụ thể. \\
\hline
\textbf{Tác nhân tham gia} & Quản trị viên hệ thống \\
\hline
\textbf{Luồng sự kiện chính} & 
\begin{enumerate}
    \vspace{-18pt}
    \item Admin điều hướng đến trang "Quản lý Người dùng" và chọn một người dùng.
    \item Admin chọn tab/mục "Phân quyền".
    \item Admin chọn "Thêm vai trò".
    \item Admin chọn một vai trò từ danh sách (ví dụ: "Warehouse Manager").
    \item Nếu vai trò có phạm vi "BRANCH", Admin phải chọn (các) chi nhánh áp dụng.
    \item HNếu vai trò có phạm vi "GLOBAL", Admin chỉ cần gán (Assign Organization-Wide Role).
    \item Admin nhấn "Lưu".
    \item Hệ thống tạo bản ghi trong bảng user_role_assignments.
\end{enumerate} \\
\hline
\textbf{Luồng sự kiện khác} & 
\begin{enumerate}
    \vspace{-18pt}
    \item[3a.] Admin xóa một vai trò đã gán khỏi người dùng.
\end{enumerate} \\
\hline
\textbf{Yêu cầu đặc biệt} & Logic kiểm tra quyền (Permission Check Logic) phải kết hợp cả quyền GLOBAL và quyền BRANCH (nếu branch_id khớp). \\
\hline
\textbf{Trạng thái hệ thống ngay khi bắt đầu thực hiện} & Admin đã đăng nhập và có quyền "ASSIGN_ROLES". Người dùng và Vai trò đã tồn tại. \\
\hline
\textbf{Trạng thái hệ thống sau khi bắt đầu thực hiện} & Người dùng được gán (hoặc bị gỡ) vai trò. \\
\hline
\textbf{Điểm mở rộng} & Không có \\
\hline
\end{tabular}
