\subsection{Nghiệp vụ kiểm kê kho}

\begin{itemize}
    \item \textbf{Lập kế hoạch kiểm kê:}
    \begin{itemize}
        \item Xác định mục đích, phạm vi và thời điểm kiểm kê.
        \item Phân công trách nhiệm cho các bộ phận và cá nhân tham gia.
        \item Chuẩn bị các công cụ và tài liệu cần thiết (bảng kê, bút, máy tính, phần mềm...).
        \item Thông báo trước cho các bộ phận liên quan để phối hợp.
    \end{itemize}
    
    \item \textbf{Chuẩn bị kho bãi:}
    \begin{itemize}
        \item Sắp xếp hàng hóa gọn gàng, đảm bảo lối đi thông thoáng.
        \item Chuẩn bị khu vực tập kết hàng hóa sau kiểm kê (nếu cần).
        \item Đảm bảo điều kiện ánh sáng tốt để dễ dàng quan sát.
    \end{itemize}
    
    \item \textbf{Tiến hành kiểm kê:}
    \begin{itemize}
        \item Đếm số lượng thực tế của từng loại hàng hóa có trong kho.
        \item Ghi chép chính xác số lượng đã đếm vào bảng kê hoặc phần mềm.
        \item Đối chiếu số lượng thực tế với số liệu trên sổ sách hoặc phần mềm quản lý kho.
        \item Lập biên bản ghi nhận số lượng kiểm kê thực tế.
    \end{itemize}
    
    \item \textbf{Xử lý sai lệch:}
    \begin{itemize}
        \item Xác định nguyên nhân gây ra sự khác biệt giữa số lượng thực tế và số liệu trên hệ thống.
        \item Điều chỉnh số liệu trên hệ thống cho khớp với số lượng thực tế đã kiểm kê.
        \item Lập biên bản giải trình và đề xuất biện pháp xử lý đối với các trường hợp sai lệch lớn (ví dụ: mất mát, hư hỏng).
    \end{itemize}
    
    \item \textbf{Lập báo cáo kiểm kê:}
    \begin{itemize}
        \item Tổng hợp kết quả kiểm kê.
        \item Phân tích các sai lệch và đưa ra các nhận xét, đánh giá.
        \item Đề xuất các biện pháp cải thiện quy trình quản lý kho để giảm thiểu sai sót trong tương lai.
    \end{itemize}
    
    \item \textbf{Lưu trữ hồ sơ kiểm kê:}
    \begin{itemize}
        \item Lưu trữ các biên bản kiểm kê, bảng kê chi tiết và báo cáo kiểm kê một cách cẩn thận để phục vụ cho công tác kiểm tra sau này.
    \end{itemize}
\end{itemize}

Nghiệp vụ kiểm kê kho là một hoạt động quan trọng giúp doanh nghiệp nắm bắt chính xác tình hình tồn kho, phát hiện và xử lý các vấn đề liên quan đến hàng hóa, từ đó nâng cao hiệu quả quản lý và giảm thiểu rủi ro.
